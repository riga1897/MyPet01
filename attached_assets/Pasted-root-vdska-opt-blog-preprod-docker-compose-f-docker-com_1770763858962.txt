global
    log stdout format raw local0
    maxconn 4096
    tune.ssl.default-dh-param 2048

defaults
    log global
    mode tcp
    option tcplog
    option dontlognull
    timeout connect 10s
    timeout client 60s
    timeout server 60s
    retries 3

frontend ft_ssl
    bind *:443
    mode tcp
    option tcplog

    tcp-request inspect-delay 5s
    tcp-request content accept if { req_ssl_hello_type 1 }

    acl is_vpn req_ssl_sni -i vpn.mine-craft.su
    acl is_web req_ssl_sni -i www.mine-craft.su
    acl is_site req_ssl_sni -i site.mine-craft.su

    use_backend bk_vpn if is_vpn
    use_backend bk_web if is_web
    use_backend bk_web if is_site

    default_backend bk_web

frontend ft_http
    bind *:80
    mode http
    option httplog

    acl is_acme path_beg /.well-known/acme-challenge/

    use_backend bk_certbot if is_acme
    default_backend bk_http_redirect

backend bk_web
    mode tcp
    balance roundrobin
    option tcp-check
    server nginx 127.0.0.1:8443 check

backend bk_vpn
    mode tcp
    balance roundrobin
    option tcp-check
    server softether 127.0.0.1:4443 check

backend bk_http_redirect
    mode http
    http-request redirect scheme https code 301

backend bk_certbot
    mode http
    server certbot 127.0.0.1:8080 check

frontend ft_minecraft
    bind *:25565
    mode tcp
    option tcplog
    default_backend bk_minecraft

frontend ft_minecraft_rcon
    bind *:25575
    mode tcp
    option tcplog
    default_backend bk_minecraft_rcon

backend bk_minecraft
    mode tcp
    balance roundrobin
    option tcp-check

    # Основной сервер (через VPN) - проверка каждые 5 сек
    server newnout01 newnout01:25565 check inter 5s fall 3 rise 2 init-addr last,libc,none

    # Резервный сервер (внешний) - активируется если основной недоступен
    server backup mainserv01.netcraze.pro:25565 check backup

backend bk_minecraft_rcon
    mode tcp
    balance roundrobin
    option tcp-check

    # Основной сервер (через VPN)
    server newnout01 newnout01:25575 check inter 5s fall 3 rise 2 init-addr last,libc,none

    # Резервный сервер (внешний)
    server backup mainserv01.netcraze.pro:25575 check backup

listen stats
    bind 127.0.0.1:8404
    mode http
    stats enable
    stats uri /stats
    stats refresh 10s
    stats admin if LOCALHOST
